Функция ПолучитьРольПользователя(Логин, Пароль)
    // Функция ищет пользователя по логину, проверяет пароль и возвращает структуру с результатом
    Результат = Новый Структура("Роль, Пользователь, Ошибка, Сообщение");
    Результат.Роль = Неопределено;
    Результат.Пользователь = Справочники.Пользователи.ПустаяСсылка();
    Результат.Ошибка = Истина;
    
    // Поиск пользователя по логину (точное совпадение)
    НайденныйПользователь = Справочники.Пользователи.НайтиПоНаименованию(Логин, Истина);
    Если НайденныйПользователь.Пустая() Тогда
        Результат.Сообщение = "Пользователь не найден";
        Возврат Результат;
    КонецЕсли;
    
    // Сохраняем ссылку на найденного пользователя
    Результат.Пользователь = НайденныйПользователь.Ссылка;
    
    // Проверка пароля
    Если НайденныйПользователь.Пароль <> Пароль Тогда
        // Неверный пароль - увеличиваем счетчик попыток
        УвеличитьСчетчикПопытокИЗаблокироватьПриНеобходимости(Результат.Пользователь);
        Результат.Сообщение = "Неверный пароль";
        Возврат Результат;
    КонецЕсли;
    
    // Проверка активности (не заблокирован ли)
    Если Не НайденныйПользователь.Активность Тогда
        Результат.Сообщение = "Пользователь заблокирован";
        Возврат Результат;
    КонецЕсли;
    
    // Все проверки пройдены
    Результат.Роль = НайденныйПользователь.Роль;
    Результат.Ошибка = Ложь;
    Результат.Сообщение = "Успешно";
    
    Возврат Результат;
КонецФункции

Процедура УвеличитьСчетчикПопытокИЗаблокироватьПриНеобходимости(Пользователь)
    // Получаем объект пользователя для изменения
    ПользовательОбъект = Пользователь.ПолучитьОбъект();
    
    // Увеличиваем счетчик неудачных попыток
    ПользовательОбъект.КоличествоПопыток = ПользовательОбъект.КоличествоПопыток + 1;
    
    // Если превышен лимит (3 попытки) – блокируем пользователя
    Если ПользовательОбъект.КоличествоПопыток > 3 Тогда
        ПользовательОбъект.Активность = Ложь;
    КонецЕсли;
    
    // Сохраняем изменения
    ПользовательОбъект.Записать();
КонецПроцедуры

        // Если условие выполняется – пользователь правильно установил ползунок
        // Устанавливаем картинки в правильном порядке (эталонная последовательность)
